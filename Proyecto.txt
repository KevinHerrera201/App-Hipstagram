SELECT 
    u.IdUsuario, 
    u.Usuario, 
    u.PasswordHash, 
    r.NombreRol,
    r.Activo AS RolActivo,
    u.Activo AS UsuarioActivo
FROM Usuarios u
INNER JOIN Roles r ON u.IdRol = r.IdRol
WHERE u.Usuario = 'admin';
GO
